79  			}""";

81  	final var requestEntityUpdateTesseract = new HttpEntity<String>(json, csrfHeaders());
82  	final var responseUpdateTesseract = restTemplate.postForEntity(HTTP_127_0_0_1 + managementPort + "/actuator/tesseract", requestEntityUpdateTesseract, String.class);
83  
84  	assertThat(responseUpdateTesseract.getStatusCode()) //
85  			.as("Update tesseract's configs with wrong values") //

88  	// ---------------------------------------------------------------------------------------------------------------

90  	headers.setContentType(MediaType.MULTIPART_FORM_DATA);

92  	headers.set("Cookie", cookies.stream().collect(joining(";")));

95

97  			.name("file") //
98  			.filename(fileName) //
99  			.build();
