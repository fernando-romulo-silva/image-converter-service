100     Stream<Arguments> createValidImageConversionData() throws IOException {
101 
102 	final var fileName = mockMultipartFile.getOriginalFilename();
103 	final var fileBytes = mockMultipartFile.getBytes();
104 
105 	return Stream.of( //
106 			Arguments.of(fileName, fileBytes, WEB, false, null, null, null, null), //
107 			Arguments.of(fileName, fileBytes, WS, true, 885, 1417, 1426, 57) //
108 	);
109     }
110 
111     @ParameterizedTest(name = "Pos {index} : fileName ''{0}'', type ''{2}'' ")
112     @MethodSource("createValidImageConversionData")
113     @Order(3)
114     @DisplayName("Test the imageConversion's creation")
115     void createValidImageConversionTest( //
116 		    // given
117 		    final String fileName, final byte[] fileContent, final ExecutionType executionType, //
118 		    final boolean area, final Integer xAxis, final Integer yAxis, final Integer width, final Integer height) {
119 
120 	// when
121 	final var imageConversionBuilder = new ImageConversion.Builder().with($ -> {
122 	    $.fileName = fileName;
123 	    $.fileContent = fileContent;
124 	    $.executionType = executionType;
125 	    $.xAxis = xAxis;
126 	    $.yAxis = yAxis;
127 	    $.width = width;
128 	    $.height = height;
129 	});
130 
131 	final var imageConversion = imageConversionBuilder.build();
132 
133 	final var fileExtension = getExtension(fileName);
134 
135 	// then
136 	assertThat(imageConversion) //
137 			.as(format("Check the fileName ''{0}'', executionType ''{1}'', area ''{2}'' and fileExtension ''{3}''", fileName, executionType, area, fileExtension)) //
138 			.extracting("fileName", "type", "area") //
139 			.containsExactly(fileName, executionType, area) //
140 			.extracting(extension -> imageConversion.getFileType().getExtension()) //
141 			.containsAnyOf(fileExtension);
142 
143     }
